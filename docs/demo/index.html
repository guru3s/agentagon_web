Step 1 — The Problem

The world without Agentagon

Here's what your company would need to build to accept payments from AI agents. Three competing protocols, each with different verification logic.

x402 Protocol

On-chain USDC verification on Base/Solana. Requires wallet setup, signature decoding, and chain-specific RPC calls.

MPP (Stripe)

Machine Payments Protocol via Tempo. Different auth flow, different token format, different settlement rails.

ACP / AP2

Agent Commerce Protocol. Yet another verification scheme, header format, and dispute resolution path.

from web3 import Web3
from eth_account.messages import encode_defunct
import json, base64, time

def verify_x402_payment(request, price_usd):
    # Extract payment header
    payment_header = request.headers.get("X-Payment")
    if not payment_header:
        return make_402_response(price_usd)

    # Decode and parse payment proof
    try:
        payload = json.loads(base64.b64decode(payment_header))
    except Exception:
        raise PaymentError("Malformed payment header")

    # Verify on-chain transaction
    w3 = Web3(Web3.HTTPProvider(RPC_URL))
    tx = w3.eth.get_transaction(payload["tx_hash"])
    if not tx or tx["to"].lower() != USDC_CONTRACT.lower():
        raise PaymentError("Invalid transaction")

    # Verify amount matches price
    decoded = decode_transfer(tx["input"])
    if decoded["amount"] < price_to_usdc(price_usd):
        raise PaymentError("Insufficient payment")

    # Verify recipient is our wallet
    if decoded["to"].lower() != OUR_WALLET.lower():
        raise PaymentError("Wrong recipient")

    # Verify not replayed
    if tx_already_used(payload["tx_hash"]):
        raise PaymentError("Replay detected")
    mark_tx_used(payload["tx_hash"])

    # Log to ledger (separate system)
    log_transaction(agent=payload["sender"], amount=price_usd,
                    endpoint=request.path, protocol="x402")

    return True
